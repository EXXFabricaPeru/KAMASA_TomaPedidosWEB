CREATE PROCEDURE "EXX_TPED_Promociones_Buscar"
(
	IN P_CodigoOperacion varchar(100)
)
AS
BEGIN
	
	SELECT	U_EXP_CODOPERACION,
			U_EXP_LINEID, 
			U_EXP_ITEMCODE, 
			U_EXP_CANTIDAD, 
			U_EXP_PRECIO, 
			U_EXP_PORCENT, 
			U_EXP_PRECIO2, 
			U_EXP_LINETOTAL, 
			U_EXP_IMPUESTO, 
			U_EXP_TOTAL, 
			U_EXP_PROMOCION, 
			U_EXP_TIPO, 
			IFNULL(U_EXP_VALOR, 0) "U_EXP_VALOR", 
			IFNULL(U_EXP_ASIGNAR, 0) "U_EXP_ASIGNAR", 
			U_EXP_COLOR, 
			U_EXP_TIPODSCTO, 
			IFNULL(U_EXP_APLICACANT, 0) "U_EXP_APLICACANT", 
			U_EXP_REFERENCIA
	FROM "@EXP_PROMO_TEMP" 
	WHERE U_EXP_CODOPERACION = :P_CodigoOperacion;

END
